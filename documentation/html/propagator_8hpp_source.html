117  virtual void setDensity(const std::vector<FLOAT_T>& radii_, const std::vector<FLOAT_T>& rhos_){
118 
119  if(rhos_.size() != radii_.size()){
120  throw std::runtime_error("setDensity : rhos.size() != radii.size()");
121  }
122 
123  if(rhos_.size() == 0 || radii_.size() == 0){
124  throw std::runtime_error("setDensity : vectors must not be empty");
125  }
126 
127  bool needFlip = false;
128 
129  if(radii_.size() >= 2){
130  int sign = (radii_[1] - radii_[0] > 0 ? 1 : -1);
131 
132  for(size_t i = 1; i < radii_.size(); i++){
133  if((radii_[i] - radii_[i-1]) * sign < 0)
134  throw std::runtime_error("radii order messed up");
135  }
136 
137  if(sign == 1)
138  needFlip = true;
139  }
140 
141  radii = radii_;
142  rhos = rhos_;
143 
144  if(needFlip){
145  std::reverse(radii.begin(), radii.end());
146  std::reverse(rhos.begin(), rhos.end());
147  }
148 
149  coslimit.clear();
150 
151  // first element of _Radii is largest radius
152  for(size_t i=0; i < radii.size() ; i++ )
153  {
154  // Using a cosine threshold
155  FLOAT_T x = -1* sqrt( 1 - (radii[i] * radii[i] / ( Constants<FLOAT_T>::REarth()*Constants<FLOAT_T>::REarth())) );
156  if ( i == 0 ) x = 0;
157  coslimit.push_back(x);
158  }
159 
160  setMaxlayers();
161  }

191  virtual void setMNSMatrix(FLOAT_T theta12, FLOAT_T theta13, FLOAT_T theta23, FLOAT_T dCP){
192 
193  const FLOAT_T s12 = sin(theta12);
194  const FLOAT_T s13 = sin(theta13);
195  const FLOAT_T s23 = sin(theta23);
196  const FLOAT_T c12 = cos(theta12);
197  const FLOAT_T c13 = cos(theta13);
198  const FLOAT_T c23 = cos(theta23);
199 
200  const FLOAT_T sd = sin(dCP);
201  const FLOAT_T cd = cos(dCP);
202 
203  U(0,0).re = c12*c13;
204  U(0,0).im = 0.0;
205  U(0,1).re = s12*c13;
206  U(0,1).im = 0.0;
207  U(0,2).re = s13*cd;
208  U(0,2).im = -s13*sd;
209  U(1,0).re = -s12*c23-c12*s23*s13*cd;
210  U(1,0).im = -c12*s23*s13*sd;
211  U(1,1).re = c12*c23-s12*s23*s13*cd;
212  U(1,1).im = -s12*s23*s13*sd;
213  U(1,2).re = s23*c13;
214  U(1,2).im = 0.0;
215  U(2,0).re = s12*s23-c12*c23*s13*cd;
216  U(2,0).im = -c12*c23*s13*sd;
217  U(2,1).re = -c12*s23-s12*c23*s13*cd;
218  U(2,1).im = -s12*c23*s13*sd;
219  U(2,2).re = c23*c13;
220  U(2,2).im = 0.0;
221  }

226  virtual void setNeutrinoMasses(FLOAT_T dm12sq, FLOAT_T dm23sq){
227  FLOAT_T mVac[3];
228 
229  mVac[0] = 0.0;
230  mVac[1] = dm12sq;
231  mVac[2] = dm12sq + dm23sq;
232 
233  const FLOAT_T delta = 5.0e-9;
234  /* Break any degeneracies */
235  if (dm12sq == 0.0) mVac[0] -= delta;
236  if (dm23sq == 0.0) mVac[2] += delta;
237 
238  DM(0,0) = 0.0;
239  DM(1,1) = 0.0;
240  DM(2,2) = 0.0;
241  DM(0,1) = mVac[0]-mVac[1];
242  DM(1,0) = -DM(0,1);
243  DM(0,2) = mVac[0]-mVac[2];
244  DM(2,0) = -DM(0,2);
245  DM(1,2) = mVac[1]-mVac[2];
246  DM(2,1) = -DM(1,2);
247 
248  }
